    #!/bin/bash

#~/.bash_zc/supp/ZCSupport_pod_pre示例

projectName=$(basename $1)
if [[ "$projectName" == "Gome"* ]] && [ -f "$(dirname $1)/setupPod.command" ]; then
 sh $(dirname $1)/setupPod.command
 # 不继续执行pod install
 return 1
fi

return 0
